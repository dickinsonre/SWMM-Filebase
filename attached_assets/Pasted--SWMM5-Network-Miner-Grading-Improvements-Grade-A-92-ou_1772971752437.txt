  function detectVersions(files) {
    // Group files by base name (strip version suffixes)
    const groups = {};
    for (const file of files) {
      const base = file.filename
        .replace(/_v\d+/i, '')
        .replace(/_\d{8}/, '')
        .replace(/_rev\d+/i, '');
      if (!groups[base]) groups[base] = [];
      groups[base].push(file);
    }
    
    // Sort each group by creation date
    for (const base in groups) {
      groups[base].sort((a, b) => 
        new Date(a.createdAt) - new Date(b.createdAt));
    }
    
    return Object.entries(groups)
      .filter(([, files]) => files.length > 1);
  }

  │     [Compare Side-by-Side]                            │
  │                                                       │
  │  SIMILARITY DIMENSIONS:                               │
  │  ├── Size: node/link/subcatchment counts              │
  │  ├── Type: sections present (LID, WQ, Snow, etc.)    │
  │  ├── Complexity: links/node ratio, max pipe diameter  │
  │  ├── Parameters: Manning's n distribution, slopes     │
  │  └── Structure: shape distribution, offset patterns   │
  └──────────────────────────────────────────────────────┘

IMPLEMENTATION:
  function computeSimilarity(modelA, modelB) {
    // Normalize each dimension to 0-1
    const sizeSim = 1 - Math.abs(
      modelA.nodeCount - modelB.nodeCount) / 
      Math.max(modelA.nodeCount, modelB.nodeCount);
    
    const linkSim = 1 - Math.abs(
      modelA.linkCount - modelB.linkCount) /
      Math.max(modelA.linkCount, modelB.linkCount);
    
    const sectionSim = jaccard(
      modelA.sections, modelB.sections);
    
    // Weighted average
    return 0.3 * sizeSim + 0.3 * linkSim + 
           0.2 * sectionSim + 0.2 * complexitySim;
  }
