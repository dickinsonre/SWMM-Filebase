  Build a client-side .inp parser:
  
  function parseINP(content) {
    const sections = {};
    let currentSection = null;
    
    for (const line of content.split('\n')) {
      const trimmed = line.trim();
      
      // Skip comments and blank lines
      if (trimmed === '' || trimmed.startsWith(';;')) continue;
      
      // Section header
      if (trimmed.startsWith('[') && trimmed.endsWith(']')) {
        currentSection = trimmed.slice(1, -1).toUpperCase();
        sections[currentSection] = [];
        continue;
      }
      
      // Section content
      if (currentSection) {
        sections[currentSection].push(trimmed);
      }
    }
    
    return {
      sections: sections,
      sectionCount: Object.keys(sections).length,
      junctions: (sections['JUNCTIONS'] || []).length,
      outfalls: (sections['OUTFALLS'] || []).length,
      storage: (sections['STORAGE'] || []).length,
      conduits: (sections['CONDUITS'] || []).length,
      pumps: (sections['PUMPS'] || []).length,
      orifices: (sections['ORIFICES'] || []).length,
      weirs: (sections['WEIRS'] || []).length,
      subcatchments: (sections['SUBCATCHMENTS'] || []).length,
      raingages: (sections['RAINGAGES'] || []).length,
      totalNodes: (sections['JUNCTIONS'] || []).length +
                  (sections['OUTFALLS'] || []).length +
                  (sections['STORAGE'] || []).length,
      totalLinks: (sections['CONDUITS'] || []).length +
                  (sections['PUMPS'] || []).length +
                  (sections['ORIFICES'] || []).length +
                  (sections['WEIRS'] || []).length,
    };
  }

  This is ~50 lines of JavaScript. It runs in the browser.
  It parses any .inp file instantly.
